#Philippines Initialization Event - fires after 1 day
country_event = {
	id = phi_init.1
	title = phi_init.1.t
	desc = phi_init.1.d
	picture = generic_political_discussion

	is_triggered_only = yes

	trigger = {
		original_tag = PHI
		has_country_flag = is_corruption_nation
	}

	option = {
		name = phi_init.1.a
		log = "[GetDateText]: [This.GetName]: phi_init.1.a executed"
		PHI_update_corruption_effects = yes
	}
}

#Trigger initialization event on game load
country_event = {
	id = phi_init.0
	hidden = yes
	
	is_triggered_only = yes
	
	immediate = {
		log = "[GetDateText]: [Root.GetName]: phi_init.0"
		if = {
			limit = {
				original_tag = PHI
				has_country_flag = is_corruption_nation
			}
			country_event = { id = phi_init.1 days = 1 }
		}
	}
}

##Standardized Event Idea Swappers (triggers every week or so)
#Low Stability good
country_event = {
	id = milf.1
	immediate = {
		log = "[GetDateText]: [Root.GetName]: event milf.1"
		remove_ideas = PHI_milf_high_effect
		remove_ideas = PHI_milf_medium_effect
		add_ideas = PHI_milf_low_effect
	}
	mean_time_to_happen = {
		days = 7
	}
	trigger = {
		check_variable = { PHI_strength_of_milf < 20 }
		has_country_flag = phi_milf_exists
	}
	hidden = yes
}

##Corruption Events

country_event = {
	id = corruption_event.1
	title = corruption_event.1.t
	desc = corruption_event.1.d
	picture = corruption

	is_triggered_only = yes

	option = {
		name = corruption_event.1.a
		log = "[GetDateText]: [This.GetName]: corruption_event.1.a executed"
		set_variable = { temp_corruption_change = 5 }
		PHI_change_corruption = yes
		add_stability = -0.02
	}
	option = {
		name = corruption_event.1.b
		log = "[GetDateText]: [This.GetName]: corruption_event.1.b executed"
		set_variable = { temp_corruption_change = -3 }
		PHI_change_corruption = yes

	}
}
