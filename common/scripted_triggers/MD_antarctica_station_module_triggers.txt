# Use this compact default to avoid enormous tooltip expansion from nested always=yes checks.

antarctica_station_install_module_unlocked = {
	always = yes
}

antarctica_station_slot_unlocked_by_tier = {
	custom_trigger_tooltip = {
		tooltip = ANTARCTICA_STATION_INSTALL_MODULE_SLOT_UNLOCK_TT
		OR = {
			AND = {
				check_variable = { global.selected_antarctica_station_tier > 0 }
				OR = {
					check_variable = { global.selected_antarctica_station_selected_slot = 5 }
					check_variable = { global.selected_antarctica_station_selected_slot = 7 }
					check_variable = { global.selected_antarctica_station_selected_slot = 8 }
					check_variable = { global.selected_antarctica_station_selected_slot = 11 }
					check_variable = { global.selected_antarctica_station_selected_slot = 12 }
					check_variable = { global.selected_antarctica_station_selected_slot = 14 }
				}
			}
			AND = {
				check_variable = { global.selected_antarctica_station_tier > 1 }
				OR = {
					check_variable = { global.selected_antarctica_station_selected_slot = 1 }
					check_variable = { global.selected_antarctica_station_selected_slot = 2 }
					check_variable = { global.selected_antarctica_station_selected_slot = 6 }
				}
			}
			AND = {
				check_variable = { global.selected_antarctica_station_tier > 2 }
				OR = {
					check_variable = { global.selected_antarctica_station_selected_slot = 4 }
					check_variable = { global.selected_antarctica_station_selected_slot = 9 }
					check_variable = { global.selected_antarctica_station_selected_slot = 10 }
					check_variable = { global.selected_antarctica_station_selected_slot = 13 }
				}
			}
		}
	}
}

antarctica_station_selected_slot_module_different = {
	custom_trigger_tooltip = {
		tooltip = ANTARCTICA_STATION_INSTALL_MODULE_DIFFERENT_TT
		OR = {
			check_variable = { global.selected_antarctica_station_selected_slot_installed_module < 1 }
			NOT = { check_variable = { global.selected_antarctica_station_selected_slot_installed_module = global.selected_antarctica_station_selected_module } }
		}
	}
}

# Usage: Check if installing selected module will maintain positive power balance
antarctica_can_install_module_power_balance = {
	custom_trigger_tooltip = {
		tooltip = ANTARCTICA_STATION_INSTALL_MODULE_POWER_BALANCE_TT
		OR = {
			check_variable = { global.selected_antarctica_station_preview_output_power > 0 }
			check_variable = { global.selected_antarctica_station_preview_output_power = 0 }
		}
	}
}

# Usage: Check if installing selected module will maintain sufficient staff availability
# preview_required_staff represents net available staff after swapping modules.
# Negative value means insufficient staff capacity.
antarctica_can_install_module_staff_availability = {
	custom_trigger_tooltip = {
		tooltip = ANTARCTICA_STATION_INSTALL_MODULE_STAFF_BALANCE_TT
		OR = {
			check_variable = { global.selected_antarctica_station_preview_output_required_staff > 0 }
			check_variable = { global.selected_antarctica_station_preview_output_required_staff = 0 }
		}
	}
}

# Usage: Info trigger for showing Antarctic winter seasonal effects
# April-October = winter = 2x cost and time multiplier
antarctica_station_install_module_winter_penalty = {
	custom_trigger_tooltip = {
		tooltip = ANTARCTICA_STATION_INSTALL_MODULE_WINTER_PENALTY_TT
		OR = {
			check_variable = { global.month = 4 }
			check_variable = { global.month = 5 }
			check_variable = { global.month = 6 }
			check_variable = { global.month = 7 }
			check_variable = { global.month = 8 }
			check_variable = { global.month = 9 }
			check_variable = { global.month = 10 }
		}
	}
}

# Usage: Called via meta_trigger with ID resolved to the station id from global.antarctica_station_ids^N.
# Uses country-capital continent and the active Antarctica region filters.
antarctica_dynamic_station_matches_selected_filter = {
	OR = {
		AND = {
			check_variable = { station_filter_america > 1 }
			OR = {
				var:global.research_station_[ID]_controller = { capital_scope = { is_on_continent = north_america } }
				var:global.research_station_[ID]_controller = { capital_scope = { is_on_continent = south_america } }
			}
		}
		AND = {
			check_variable = { station_filter_asia > 1 }
			OR = {
				var:global.research_station_[ID]_controller = { capital_scope = { is_on_continent = asia } }
				var:global.research_station_[ID]_controller = { capital_scope = { is_on_continent = australia } }
			}
		}
		AND = {
			check_variable = { station_filter_oceania > 1 }
			OR = {
				var:global.research_station_[ID]_controller = { capital_scope = { is_on_continent = asia } }
				var:global.research_station_[ID]_controller = { capital_scope = { is_on_continent = australia } }
			}
		}
		AND = {
			check_variable = { station_filter_europe > 1 }
			var:global.research_station_[ID]_controller = { capital_scope = { is_on_continent = europe } }
		}
		AND = {
			check_variable = { station_filter_africa > 1 }
			OR = {
				var:global.research_station_[ID]_controller = { capital_scope = { is_on_continent = africa } }
				var:global.research_station_[ID]_controller = { capital_scope = { is_on_continent = middle_east } }
			}
		}
	}
}
