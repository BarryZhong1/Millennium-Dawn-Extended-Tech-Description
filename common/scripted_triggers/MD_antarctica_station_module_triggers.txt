antarctica_station_install_module_unlocked = {
	always = yes
}

antarctica_station_slot_unlocked_by_tier = {
	OR = {
		AND = {
			check_variable = { global.selected_antarctica_station_tier > 0 }
			OR = {
				check_variable = { global.selected_antarctica_station_selected_slot = 5 }
				check_variable = { global.selected_antarctica_station_selected_slot = 7 }
				check_variable = { global.selected_antarctica_station_selected_slot = 8 }
				check_variable = { global.selected_antarctica_station_selected_slot = 11 }
				check_variable = { global.selected_antarctica_station_selected_slot = 12 }
				check_variable = { global.selected_antarctica_station_selected_slot = 14 }
			}
		}
		AND = {
			check_variable = { global.selected_antarctica_station_tier > 1 }
			OR = {
				check_variable = { global.selected_antarctica_station_selected_slot = 1 }
				check_variable = { global.selected_antarctica_station_selected_slot = 2 }
				check_variable = { global.selected_antarctica_station_selected_slot = 6 }
			}
		}
		AND = {
			check_variable = { global.selected_antarctica_station_tier > 2 }
			OR = {
				check_variable = { global.selected_antarctica_station_selected_slot = 4 }
				check_variable = { global.selected_antarctica_station_selected_slot = 9 }
				check_variable = { global.selected_antarctica_station_selected_slot = 10 }
				check_variable = { global.selected_antarctica_station_selected_slot = 13 }
			}
		}
	}
}

antarctica_station_selected_slot_module_different = {
	OR = {
		check_variable = { global.selected_antarctica_station_selected_slot_installed_module < 1 }
		NOT = { check_variable = { global.selected_antarctica_station_selected_slot_installed_module = global.selected_antarctica_station_selected_module } }
	}
}

# Usage: Check if installing selected module will maintain positive power balance
antarctica_can_install_module_power_balance = {
	OR = {
		check_variable = { global.selected_antarctica_station_preview_output_power > 0 }
		check_variable = { global.selected_antarctica_station_preview_output_power = 0 }
	}
}

# Usage: Check if installing selected module will maintain sufficient staff availability
# preview_required_staff represents net available staff with selected module (housing capacity - staff needed)
# Negative value means insufficient staff capacity
antarctica_can_install_module_staff_availability = {
	OR = {
		check_variable = { global.selected_antarctica_station_preview_output_required_staff > 0 }
		check_variable = { global.selected_antarctica_station_preview_output_required_staff = 0 }
	}
}
