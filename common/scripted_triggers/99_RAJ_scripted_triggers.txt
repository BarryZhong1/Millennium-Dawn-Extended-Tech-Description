BRICS_valid_invite_target = {
	exists = yes
	NOT = { tag = ROOT }
	NOT = { is_in_faction_with = ROOT }
	NOT = { has_country_flag = BRICS_full_member_status }
	NOT = { has_country_flag = BRICS_associate_status }
	NOT = { has_country_flag = BRICS_observer_status }
}

# Evaluate in target country scope; ROOT = BRICS leader
BRICS_can_full_member = {
	custom_trigger_tooltip = {
		tooltip = BRICS_full_member_requirements_tt
		influence_higher_50 = yes
		has_opinion = { target = ROOT value > 100 }
		OR = {
			num_of_factories > 25
			has_idea = economic_boom
		}
		NOT = { has_government = fascism }
	}
}

# Evaluate in target country scope; ROOT = BRICS leader
BRICS_can_associate_member = {
	custom_trigger_tooltip = {
		tooltip = BRICS_associate_member_requirements_tt
		influence_higher_40 = yes
		has_opinion = { target = ROOT value > 50 }
		OR = {
			has_government = democratic
			has_government = neutrality
			has_government = communism
		}
	}
}

# Evaluate in target country scope; ROOT = BRICS leader
BRICS_can_observer = {
	custom_trigger_tooltip = {
		tooltip = BRICS_observer_requirements_tt
		influence_higher_25 = yes
		has_opinion = { target = ROOT value > 0 }
	}
}

RAJ_rebel_opinion_collapsed = {
	if = {
		limit = { check_variable = { RAJ_rebel_target_state = 993 } }
		check_variable = { RAJ_muslim_opinion < 5 }
	}
	else_if = {
		limit = {
			OR = {
				check_variable = { RAJ_rebel_target_state = 980 }
				check_variable = { RAJ_rebel_target_state = 477 }
			}
		}
		check_variable = { RAJ_christians_opinion < 5 }
	}
	else_if = {
		limit = { check_variable = { RAJ_rebel_target_state = 982 } }
		check_variable = { RAJ_buddhist_opinion < 5 }
	}
	else_if = {
		limit = { check_variable = { RAJ_rebel_target_state = 429 } }
		check_variable = { RAJ_sikh_opinion < 5 }
	}
}

RAJ_side_a_opinion_lower_than_b = {
	if = {
		limit = { check_variable = { RAJ_conflict_side_a = 1 } }
		if = {
			limit = { check_variable = { RAJ_conflict_side_b = 2 } }
			check_variable = { RAJ_hindus_opinion < RAJ_muslim_opinion }
		}
		else_if = {
			limit = { check_variable = { RAJ_conflict_side_b = 3 } }
			check_variable = { RAJ_hindus_opinion < RAJ_christians_opinion }
		}
	}
	else_if = {
		limit = { check_variable = { RAJ_conflict_side_a = 2 } }
		if = {
			limit = { check_variable = { RAJ_conflict_side_b = 1 } }
			check_variable = { RAJ_muslim_opinion < RAJ_hindus_opinion }
		}
		else_if = {
			limit = { check_variable = { RAJ_conflict_side_b = 3 } }
			check_variable = { RAJ_muslim_opinion < RAJ_christians_opinion }
		}
	}
	else_if = {
		limit = { check_variable = { RAJ_conflict_side_a = 3 } }
		if = {
			limit = { check_variable = { RAJ_conflict_side_b = 1 } }
			check_variable = { RAJ_christians_opinion < RAJ_hindus_opinion }
		}
		else_if = {
			limit = { check_variable = { RAJ_conflict_side_b = 2 } }
			check_variable = { RAJ_christians_opinion < RAJ_muslim_opinion }
		}
	}
}
