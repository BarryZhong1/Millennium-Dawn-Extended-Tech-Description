# Sinicization Scripted Effects — Tibet, Xinjiang, Inner Mongolia
# Call these effects on the SAR scope after any change to sinicization_level.
# They clamp the variable and recalculate all dynamic modifier variables.
#
# Stability:         +0.05 at level 0  →  -0.15 at level 100  (0.05 - 0.002 * level)
# PP factor:         +0.05 at level 0  →  -0.05 at level 100  (0.05 - 0.001 * level)
# Consumer goods:    +0.05 at level 0  →  -0.05 at level 100  (0.05 - 0.001 * level)
# Prod. speed:        0.00 at level 0  →  +0.10 at level 100  (0.001 * level)
# Resistance target:  0.00 at level 0  →  +0.10 at level 100  (0.001 * level)
#
# Tier thresholds (fire event only on first crossing):
#   0–32:   Tier 0 — autonomy idea
#   33–65:  Tier 1 — tension idea; fires threshold event
#   66–99:  Tier 2 — repression idea; fires repression event
#   100:    Complete — remove all ideas; fires sinicization-complete event
#
# update_CHI_SAR_network_modifiers runs on CHI scope and must be called separately
# after any SAR status change (focus completion, annexation, game start).

update_TIB_sinicization_modifiers = {
	clamp_variable = { var = sinicization_level  min = 0  max = 100 }

	# stability_factor: 0.05 - 0.002 * level
	set_temp_variable = { sin_t = sinicization_level }
	multiply_temp_variable = { sin_t = -0.002 }
	add_to_temp_variable = { sin_t = 0.05 }
	set_variable = { TIB_sin_stability = sin_t }

	# political_power_factor: 0.05 - 0.001 * level
	set_temp_variable = { sin_t = sinicization_level }
	multiply_temp_variable = { sin_t = -0.001 }
	add_to_temp_variable = { sin_t = 0.05 }
	set_variable = { TIB_sin_pp = sin_t }

	# consumer_goods_factor: 0.05 - 0.001 * level
	set_temp_variable = { sin_t = sinicization_level }
	multiply_temp_variable = { sin_t = -0.001 }
	add_to_temp_variable = { sin_t = 0.05 }
	set_variable = { TIB_sin_cg = sin_t }

	# production_speed_buildings_factor: 0.001 * level
	set_temp_variable = { sin_t = sinicization_level }
	multiply_temp_variable = { sin_t = 0.001 }
	set_variable = { TIB_sin_production = sin_t }

	# resistance_target: 0.001 * level
	set_temp_variable = { sin_t = sinicization_level }
	multiply_temp_variable = { sin_t = 0.001 }
	set_variable = { TIB_sin_resistance = sin_t }

	# Tier idea management
	if = {
		limit = { check_variable = { sinicization_level < 33 } }
		if = {
			limit = { NOT = { has_idea = TIB_dalai_lama_governance } }
			add_ideas = TIB_dalai_lama_governance
		}
		if = {
			limit = { has_idea = TIB_cultural_tension }
			remove_ideas = TIB_cultural_tension
		}
		if = {
			limit = { has_idea = TIB_forced_integration }
			remove_ideas = TIB_forced_integration
		}
	}
	else_if = {
		limit = { check_variable = { sinicization_level < 66 } }
		if = {
			limit = { NOT = { has_country_flag = TIB_sin_tier1_reached } }
			set_country_flag = TIB_sin_tier1_reached
			country_event = { id = chi_sar_sin.1 }
		}
		if = {
			limit = { has_idea = TIB_dalai_lama_governance }
			remove_ideas = TIB_dalai_lama_governance
		}
		if = {
			limit = { NOT = { has_idea = TIB_cultural_tension } }
			add_ideas = TIB_cultural_tension
		}
		if = {
			limit = { has_idea = TIB_forced_integration }
			remove_ideas = TIB_forced_integration
		}
	}
	else_if = {
		limit = { check_variable = { sinicization_level < 100 } }
		if = {
			limit = { NOT = { has_country_flag = TIB_sin_tier2_reached } }
			set_country_flag = TIB_sin_tier2_reached
			country_event = { id = chi_sar_sin.2 }
		}
		if = {
			limit = { has_idea = TIB_dalai_lama_governance }
			remove_ideas = TIB_dalai_lama_governance
		}
		if = {
			limit = { has_idea = TIB_cultural_tension }
			remove_ideas = TIB_cultural_tension
		}
		if = {
			limit = { NOT = { has_idea = TIB_forced_integration } }
			add_ideas = TIB_forced_integration
		}
	}
	else = {
		if = {
			limit = { NOT = { has_country_flag = TIB_sin_complete } }
			set_country_flag = TIB_sin_complete
			country_event = { id = chi_sar_sin.3 }
		}
		if = {
			limit = { has_idea = TIB_dalai_lama_governance }
			remove_ideas = TIB_dalai_lama_governance
		}
		if = {
			limit = { has_idea = TIB_cultural_tension }
			remove_ideas = TIB_cultural_tension
		}
		if = {
			limit = { has_idea = TIB_forced_integration }
			remove_ideas = TIB_forced_integration
		}
	}
}

# CHI SAR Network Modifier Update
# Must be called on CHI scope after any SAR subject status change.
# Resets all network variables from scratch and re-evaluates each active SAR.

update_CHI_SAR_network_modifiers = {
	# Reset all network variables
	set_variable = { CHI_sar_stability = 0 }
	set_variable = { CHI_sar_pp = 0 }
	set_variable = { CHI_sar_hkg_cg = 0 }
	set_variable = { CHI_sar_hkg_trade = 0 }
	set_variable = { CHI_sar_mac_cg = 0 }
	set_variable = { CHI_sar_tib_manpower = 0 }
	set_variable = { CHI_sar_etk_production = 0 }
	set_variable = { CHI_sar_tai_research = 0 }
	set_variable = { CHI_sar_omg_production = 0 }

	# HKG: financial hub — cheaper consumer goods, reduced trade law cost
	if = {
		limit = {
			any_subject_country = { original_tag = HKG }
		}
		add_to_variable = { CHI_sar_stability = 0.01 }
		add_to_variable = { CHI_sar_pp = 0.01 }
		set_variable = { CHI_sar_hkg_cg = -0.03 }
		set_variable = { CHI_sar_hkg_trade = -0.05 }
	}

	# MAC: services economy — additional consumer goods reduction
	if = {
		limit = {
			any_subject_country = { original_tag = MAC }
		}
		add_to_variable = { CHI_sar_stability = 0.01 }
		add_to_variable = { CHI_sar_pp = 0.01 }
		set_variable = { CHI_sar_mac_cg = -0.02 }
	}

	# TIB: frontier manpower reserve
	if = {
		limit = {
			any_subject_country = { original_tag = TIB }
		}
		add_to_variable = { CHI_sar_stability = 0.01 }
		add_to_variable = { CHI_sar_pp = 0.01 }
		set_variable = { CHI_sar_tib_manpower = 0.05 }
	}

	# ETK: resource corridor — faster construction
	if = {
		limit = {
			any_subject_country = { original_tag = ETK }
		}
		add_to_variable = { CHI_sar_stability = 0.01 }
		add_to_variable = { CHI_sar_pp = 0.01 }
		set_variable = { CHI_sar_etk_production = 0.04 }
	}

	# TAI: high-tech industry — research speed
	if = {
		limit = {
			any_subject_country = { original_tag = TAI }
		}
		add_to_variable = { CHI_sar_stability = 0.01 }
		add_to_variable = { CHI_sar_pp = 0.01 }
		set_variable = { CHI_sar_tai_research = 0.04 }
	}

	# OMG: steppe resource corridor — additional construction speed
	if = {
		limit = {
			any_subject_country = { original_tag = OMG }
		}
		add_to_variable = { CHI_sar_stability = 0.01 }
		add_to_variable = { CHI_sar_pp = 0.01 }
		set_variable = { CHI_sar_omg_production = 0.03 }
	}

	# Apply dynamic modifier if not yet present
	if = {
		limit = {
			NOT = { has_dynamic_modifier = { modifier = CHI_SAR_network_modifier } }
		}
		add_dynamic_modifier = { modifier = CHI_SAR_network_modifier }
	}
}
