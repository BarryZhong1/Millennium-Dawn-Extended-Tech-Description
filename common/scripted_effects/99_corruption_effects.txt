PHI_change_corruption = {
	custom_effect_tooltip = PHI_change_corruption_tt
	add_to_variable = { PHI_corruption_influence = temp_corruption_change }
	clamp_variable = { min = 0 max = 100 var = PHI_corruption_influence }
	if = { limit = { check_variable = { PHI_corruption_influence > 0 } check_variable = { PHI_corruption_influence < 25 } }
		remove_ideas = { PHI_idea_corruption_stage_two PHI_idea_corruption_stage_three PHI_idea_corruption_stage_four }
		add_ideas = PHI_idea_corruption_stage_one
	}
	else_if = { limit = { check_variable = { PHI_corruption_influence > 24 } check_variable = { PHI_corruption_influence < 50 } }
		remove_ideas = { PHI_idea_corruption_stage_one PHI_idea_corruption_stage_three PHI_idea_corruption_stage_four }
		add_ideas = PHI_idea_corruption_stage_two
	}
	else_if = { limit = { check_variable = { PHI_corruption_influence > 49 } check_variable = { PHI_corruption_influence < 75 } }
		remove_ideas = { PHI_idea_corruption_stage_one PHI_idea_corruption_stage_two PHI_idea_corruption_stage_four }
		add_ideas = PHI_idea_corruption_stage_three
	}
	else_if = { limit = { check_variable = { PHI_corruption_influence > 74 } }
		remove_ideas = { PHI_idea_corruption_stage_one PHI_idea_corruption_stage_two PHI_idea_corruption_stage_three }
		add_ideas = PHI_idea_corruption_stage_four
	}
	if = { limit = { NOT = { has_dynamic_modifier = { modifier = PHI_corruption_dynamic_modifier } } }
		add_dynamic_modifier = { modifier = PHI_corruption_dynamic_modifier }
	}
	force_update_dynamic_modifier = yes
}
