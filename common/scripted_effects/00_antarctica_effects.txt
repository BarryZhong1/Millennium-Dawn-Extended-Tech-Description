# set_temp_variable = { station_id = x }
# antarctica_setup_research_station = yes
antarctica_setup_research_station = {
	if = {
		limit = {
			meta_trigger = {
				text = {
					NOT = { has_global_flag = research_station_[ID]_exists }
				}
				ID = "[?station_id]"
			}
		}
		meta_effect = {
			text = {
				set_global_flag = research_station_[ID]_exists
				set_country_flag = research_station_[ID]_controller_@THIS
				set_variable = { global.research_station_[ID]_controller = THIS.id }
			}
			ID = "[?station_id]"
		}
		if = {
			limit = { check_variable = { global.selected_antarctica_station = 0 } }
			set_variable = { global.selected_antarctica_station = station_id }
			set_variable = { global.selected_antarctica_station_controller = THIS.id }
		}
		log = "[GetDateText]: [Root.GetName]: Set up research station in Antarctica. Station ID: [?station_id] with controller [Root.GetName]."
	}
	else = {
		log = "[GetDateText]: [Root.GetName]: Failed to set up research station in Antarctica. Station ID: [?station_id] already exists."
	}
}

# Usage
# set_temp_variable = { station_id = id }
# set_temp_variable = { module_category = n } # 1 powerplant, 2 life support, 3 fuel storage, 4 housing, 5 laboratories
# set_temp_variable = { module_id = n }
# antarctica_set_station_module = yes
antarctica_set_station_module = {
	if = {
		limit = {
			meta_trigger = {
				text = { has_global_flag = research_station_[ID]_exists }
				ID = "[?station_id]"
			}
		}

		if = {
			limit = { check_variable = { module_category = 1 } }
			meta_effect = {
				text = { set_variable = { global.research_station_[ID]_module_powerplant = [MOD] } }
				ID = "[?station_id]"
				MOD = "[?module_id]"
			}
		}
		else_if = {
			limit = { check_variable = { module_category = 2 } }
			meta_effect = {
				text = { set_variable = { global.research_station_[ID]_module_life_support = [MOD] } }
				ID = "[?station_id]"
				MOD = "[?module_id]"
			}
		}
		else_if = {
			limit = { check_variable = { module_category = 3 } }
			meta_effect = {
				text = { set_variable = { global.research_station_[ID]_module_fuel_storage = [MOD] } }
				ID = "[?station_id]"
				MOD = "[?module_id]"
			}
		}
		else_if = {
			limit = { check_variable = { module_category = 4 } }
			meta_effect = {
				text = { set_variable = { global.research_station_[ID]_module_housing = [MOD] } }
				ID = "[?station_id]"
				MOD = "[?module_id]"
			}
		}
		else_if = {
			limit = { check_variable = { module_category = 5 } }
			meta_effect = {
				text = { set_variable = { global.research_station_[ID]_module_laboratories = [MOD] } }
				ID = "[?station_id]"
				MOD = "[?module_id]"
			}
		}

		antarctica_recalculate_station_module_outputs = yes
		if = {
			limit = { check_variable = { global.selected_antarctica_station = station_id } }
			antarctica_set_selected_station_controller = yes
		}

		add_to_variable = { global.missile_ui_updated = 1 }
		log = "[GetDateText]: [Root.GetName]: Updated Antarctica station module. Station ID: [?station_id], Category: [?module_category], Module: [?module_id]."
	}
}

# antarctica_remove_research_station_from_display = yes
antarctica_remove_research_station_from_display = {
	set_temp_variable = { remove_index = -1 }
	for_each_loop = {
		array = global.antarctica_station_ids
		value = current_station_id
		index = i
		break = break_loop

		if = {
			limit = { check_variable = { current_station_id = station_id } }
			set_temp_variable = { remove_index = i }
			set_temp_variable = { break_loop = 1 }
		}
	}

	if = {
		limit = { check_variable = { remove_index > -1 } }
		set_variable = { global.antarctica_station_ids^remove_index = -1 }
		set_variable = { global.antarctica_station_x^remove_index = 0 }
		set_variable = { global.antarctica_station_y^remove_index = 0 }

		if = {
			limit = { check_variable = { global.selected_antarctica_station = station_id } }
			set_variable = { global.selected_antarctica_station = 0 }
			set_variable = { global.selected_antarctica_station_x = 0 }
			set_variable = { global.selected_antarctica_station_y = 0 }
		}

		log = "[GetDateText]: [Root.GetName]: Removed Antarctica station from display array. Station ID: [?station_id]."
	}
}

# set_temp_variable = { station_id = x }
# antarctica_remove_research_station = yes
antarctica_remove_research_station = {
	if = {
		limit = {
			meta_trigger = {
				text = {
					has_global_flag = research_station_[ID]_exists
					has_country_flag = research_station_[ID]_controller_@THIS
				}
				ID = "[?station_id]"
			}
		}
		meta_effect = {
			text = {
				clr_global_flag = research_station_[ID]_exists
				clr_country_flag = research_station_[ID]_controller_@THIS
				set_variable = { global.research_station_[ID]_controller = 0 }
			}
			ID = "[?station_id]"
		}
		antarctica_remove_research_station_from_display = yes
		log = "[GetDateText]: [Root.GetName]: Removed research station in Antarctica. Station ID: [?station_id] with controller [Root.GetName]."
	}
	else = {
		log = "[GetDateText]: [Root.GetName]: Failed to remove research station in Antarctica. Station ID: [?station_id] does not exist or is not controlled by the current country."
	}
}

# Usage
# set_temp_variable = { station_id = id }
# set_temp_variable = { station_x = x }
# set_temp_variable = { station_y = y }
# antarctica_research_station_position = yes
antarctica_research_station_position = {
	antarctica_initialize_station_display_arrays = yes

	if = {
		limit = {
			meta_trigger = {
				text = {
					has_global_flag = research_station_[ID]_exists
				}
				ID = "[?station_id]"
			}
		}
		meta_effect = {
			text = {
				set_variable = { global.research_station_[ID]_x = [X] }
				set_variable = { global.research_station_[ID]_y = [Y] }
			}
			ID = "[?station_id]"
			X = "[?station_x]"
			Y = "[?station_y]"
		}

		set_temp_variable = { station_found = 0 }
		for_each_loop = {
			array = global.antarctica_station_ids
			value = current_station_id
			index = i
			break = break_loop

			if = {
				limit = { check_variable = { current_station_id = station_id } }
				set_variable = { global.antarctica_station_x^i = station_x }
				set_variable = { global.antarctica_station_y^i = station_y }
				set_temp_variable = { station_found = 1 }
				set_temp_variable = { break_loop = 1 }
			}
		}

		if = {
			limit = { check_variable = { station_found = 0 } }
			add_to_array = { global.antarctica_station_ids = station_id }
			add_to_array = { global.antarctica_station_x = station_x }
			add_to_array = { global.antarctica_station_y = station_y }
		}

		if = {
			limit = {
				OR = {
					check_variable = { global.selected_antarctica_station = 0 }
					check_variable = { global.selected_antarctica_station = station_id }
				}
			}
			set_variable = { global.selected_antarctica_station = station_id }
			set_variable = { global.selected_antarctica_station_x = station_x }
			set_variable = { global.selected_antarctica_station_y = station_y }
			meta_effect = {
				text = {
					set_variable = { global.selected_antarctica_station_controller = global.research_station_[ID]_controller }
				}
				ID = "[?station_id]"
			}
		}

		set_variable = { global.missile_ui_updated = global.missile_ui_updated }
		add_to_variable = { global.missile_ui_updated = 1 }
		log = "[GetDateText]: [Root.GetName]: Updated Antarctica station position. Station ID: [?station_id], X: [?station_x], Y: [?station_y]."
	}
	else = {
		log = "[GetDateText]: [Root.GetName]: Failed to set Antarctica station position. Station ID: [?station_id] does not exist."
	}
}

# Usage
# set_temp_variable = { station_x = x }

# set_temp_variable = { station_name_id = n } # optional, for GUI name lookup later
# antarctica_create_startup_research_station = yes
antarctica_create_startup_research_station = {
	if = {
		limit = { check_variable = { global.next_antarctica_station_id < 1 } }
		set_variable = { global.next_antarctica_station_id = 1 }
	}

	set_temp_variable = { station_id = global.next_antarctica_station_id }
	add_to_variable = { global.next_antarctica_station_id = 1 }

	# If startup coordinates are omitted, spread stations diagonally as a safe fallback.
	if = {
		limit = {
			OR = {
				check_variable = { station_x < 1 }
				check_variable = { station_y < 1 }
			}
		}
		set_temp_variable = { station_offset = station_id }
		multiply_temp_variable = { station_offset = 18 }

		if = {
			limit = { check_variable = { station_x < 1 } }
			set_temp_variable = { station_x = 420 }
			add_to_temp_variable = { station_x = station_offset }
		}
		if = {
			limit = { check_variable = { station_y < 1 } }
			set_temp_variable = { station_y = 420 }
			add_to_temp_variable = { station_y = station_offset }
		}
	}

	antarctica_setup_research_station = yes
	antarctica_research_station_position = yes

	# Default station module set for newly created stations.
	meta_effect = {
		text = {
			set_variable = { global.research_station_[ID]_module_powerplant = 1 }
			set_variable = { global.research_station_[ID]_module_life_support = 1 }
			set_variable = { global.research_station_[ID]_module_fuel_storage = 1 }
			set_variable = { global.research_station_[ID]_module_housing = 1 }
			set_variable = { global.research_station_[ID]_module_laboratories = 1 }
		}
		ID = "[?station_id]"
	}

	if = {
		limit = { check_variable = { station_name_id > 0 } }
		meta_effect = {
			text = {
				set_variable = { global.research_station_[ID]_name_id = [NAME] }
			}
			ID = "[?station_id]"
			NAME = "[?station_name_id]"
		}
	}

	antarctica_recalculate_station_module_outputs = yes

	log = "[GetDateText]: [Root.GetName]: Created startup Antarctica station. Station ID: [?station_id], Name ID: [?station_name_id], X: [?station_x], Y: [?station_y]."
}
