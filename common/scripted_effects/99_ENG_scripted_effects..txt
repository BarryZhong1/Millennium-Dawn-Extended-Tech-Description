				}
				set_country_flag = ENG_moon_base_first_crisis
				country_event = { id = ENG_military.010 days = 30 random_hours = 480 } #First crisis
			}
			40 = {
				modifier = {
					factor = 0
					OR = {
						has_country_flag = ENG_moon_base_second_crisis
						NOT = {
							has_country_flag = ENG_moon_base_first_crisis

					}
				}

# Devolution Mechanic:
# Agitation = how annoying they are, basically, higher agitation means more harsher punishments for when they go against you
# -------------- HOW AGITATION WORKS -------------- #
# [-] Agitation will grow from certain focuses/events/decisions etc
# - Agitation will be lowered over time naturally
# - If you maintain a high level of agitation for a while, you'll get random events that increase nationalism
# - Coupled with random naitonalism events, you get debuffs anytime they are pissed
# - If nationalism is above 85% for too long, Agitation will start growing randomly anywhere between 1-3 every month
# - Agitation can be lowered by a lot by lower their dependency
# -------------- HOW DEPENDENCY WORKS -------------- #
# [-] Dependency is lower or increased via the focus tree (A Variable out of 100 with 3 stages)
# - In the focus tree, there will be 2 trees, 1 for pro-devolve and 1 for against per region
# - The paths are made available based on party ideology
# - Focuses will unlock decisions that count as laws that you could pass (Either pro or anti devolution laws)
# - Some of these decisions are permanent/irreversable laws, EX: Reducing Scottish parliament seats (Will get to this later)
# - Will have 1 for Scots, Welsh, and Irish
# --- Government Change in the middle of devolution ---
# [-] Gives Spirit "Promises of Devolution"
# --- What if you change governments from anti-devolution party to pro-devolution? --- #
# - After change of govt. If you already went down dickstomping devolution, it gives you an event that pops up and says:
# "Scottish seats will support you if you revert the fuck up of previous administration" (Halting Devolution). If you say accept,
# But continue devolved policies, then the Scots agitation will increase by 2X anytime they get pissed (EX: 1 agitation is now 2)
# But if you keep your promise, it will bring their agitation down and you get support from them.
# You will also get decisions to repeal certain laws that will decrease their dependency on the U.K.
# If you reject their request, you can just go down the devolution path the previous admininistration started
# If you for some reason bring back the anti-devolution party, you then get those laws (decisions) you got from focuses
# You will NOT get the decisions for the permanent decisions, this goes for both pro and anti devolution parties
# --- What if you change governments from pro-devolution party to anti-devolution? --- #
# - Gets event that lets you make a promise that you won't clamp down on progress made on devolution
# - You will be able to repeal certain laws as stated from before, as mentioned before if you decide to clamp down
# If promise is granted and you don't reverse devolution, agitation will decrease BUT it will give an equilibrium toward devolution
# If you decide to continue going the route of dick-stomping devolution, then you will start losing party popularity
# Dependency levels will decide independence levels
# - Dependent is when the constituent is in your land (not on the game map)
# - Semi - Dependent is any of the 4 autonomy levels (on the map, but they are a puppet)
# The autonomy levels will dynamically shift as the player goes deeper into devolution
# Idea is to encourage as low as possible on devolution to maximize bonuses towards the player
# Bonuses include %'s of GDP and Commonwealth nation member states which will ultimately help the U.K.
# Will also give option to force Wales & Northern Ireland to use the British Pound
# - Independent is idk take a wild guess
# -------------- HOW NATIONALISM WORKS -------------- #
# [-] Nationalism will decide whether or not the constituent will declare a referendum
# - Nationalism will grow upon doing moves that anger the constituent
# - Possible causes for nationalism are:
# Certain events/focuses/decisions that may cause harm to that constituent
# Beginning to back peddle on devolution or trying to curb it entirely
# Going to War
# - If nationalism hits 100, they will run a referendum that you then have to ratify
# Agitation will decide their reaction to your decision on referendum
# Low agitation if Rejected: non-violent protest movements (damage to economy basically)
# high agitation if rejected: Car bombings every other day & chance of revolt upon rejection (car bombings still follow)
# Low agitation if accepted: Devolve with a lot of control (This is basically outcome if u just do everything to devolve)
# High agitation if accepted: Devolve with less control (this is if they hit u w/the referendum and u accept)
# -------------- Focus Tree -------------- #
# Once someone goes down the path of either devolving or not, if they decide to nothing and edge the system, they'll get devolution fallout spirit x10
# -------------- MISC -------------- #
# [-] Add game rule that lets you spell Scotland will let you spell Scotland with 2 T's
# - If made available, all your leaders and generals get renamed to "Scott"

# Westmister Mechanic
# Main Parties: Conservatives, labour, Lib-Dem
# Smaller parties: Commie Coalition, UKIP, Greens, BNP & SNP
# Elections
# Hung Parliament if majority is not met
# Have the 650 seats and then figure out %'s of how much each party got in each election to figure out winner
# Divide 650 by party pop by each slot
# Wales and scotland: Take Party Pop, but then you take away the nationalists % Variable and then the remainder of the % left are given to the nationalist parties
# - Change the GUI for NIR to split nationalism into 3 bars, Unionism, Non-Sectarianism, & Republicanism
# - Make 3 seperate variables.
# If you go against devolution you support unionists
# If you go further devolution you can support the Republicans & Non Sectarians
# These 3 seperate variables will decide end state for NIR
# 18 NIR Seats, and take 3 variables and divide them against 18 (Based on actual variables)
# 59 Scottish (Based on actual nationalism VAR)
# 40 Welsh (Based on actual nationalism VAR)
# Whoever wins the election loses one seat as they get a speaker
# English seats: (Take Party Popularity variable) (if minor -, if major +) ( random variable selected from range of 0-10)
# (Take Party Popularity of Labour,Tories,LibDems,Greens) (then Combine) (then add Nationalist Percentage) (remainder randomly distributed)

ENG_commons_elections = {
	custom_effect_tooltip = ENG_commons_TT
	ENG_commons_seats_reset = yes
	ENG_commons_ban_parties = yes

	# --- Identify the top two parties for FPTP seat amplification ---
	# Write directly to persistent variables, then restore array before any loops.
	find_highest_in_array = {
		array = party_pop_array
		value = ENG_commons_pop1
		index = ENG_commons_idx1
	}
	set_variable = { party_pop_array^ENG_commons_idx1 = -2 }
	find_highest_in_array = {
		array = party_pop_array
		value = ENG_commons_pop2
		index = ENG_commons_idx2
	}
	# Restore winner — party_pop_array is never touched again
	set_variable = { party_pop_array^ENG_commons_idx1 = ENG_commons_pop1 }

	# --- England: 533 seats (First Past the Post, scaled) ---
	# Single loop assigns raw seat weights using VALUE comparison (matching PER parliament pattern):
	#   Winner  identified by party_pop_array^i == party_pop_array^ENG_commons_idx1 -> 2.5x weight
	#   Runner-up identified by party_pop_array^i == party_pop_array^ENG_commons_idx2 -> 1.5x weight
	#   Any other party above 5% -> 0.3x weight
	#   Below 5% -> 0 seats
	# Second loop sums all raw weights; third loop scales to exactly 533 seats.

	for_each_loop = {
		array = party_pop_array
		value = v
		index = i

		set_variable = { ENG_commons_seats^i = 0 }

		if = {
			limit = { check_variable = { party_pop_array^i = party_pop_array^ENG_commons_idx1 } }
			set_variable = { ENG_commons_seats^i = ENG_commons_pop1 }
			multiply_variable = { ENG_commons_seats^i = 2.5 }
		}
		else_if = {
			limit = { check_variable = { party_pop_array^i = party_pop_array^ENG_commons_idx2 } }
			set_variable = { ENG_commons_seats^i = ENG_commons_pop2 }
			multiply_variable = { ENG_commons_seats^i = 1.5 }
		}
		else_if = {
			limit = { check_variable = { v > 0.05 } }
			set_variable = { ENG_commons_seats^i = v }
			multiply_variable = { ENG_commons_seats^i = 0.3 }
		}
	}

	# Sum all raw weights
	set_variable = { ENG_commons_eng_seat_sum = 0 }
	for_each_loop = {
		array = party_pop_array
		value = v
		index = i
		if = {
			limit = { check_variable = { ENG_commons_seats^i > 0 } }
			add_to_variable = { ENG_commons_eng_seat_sum = ENG_commons_seats^i }
		}
	}

	# Scale to 533: seats^i = seats^i * 533 / total_raw_weight, then accumulate
	for_each_loop = {
		array = party_pop_array
		value = v
		index = i
		if = {
			limit = { check_variable = { ENG_commons_seats^i > 0 } }
			multiply_variable = { ENG_commons_seats^i = 533 }
			divide_variable = { ENG_commons_seats^i = ENG_commons_eng_seat_sum }
			round_variable = ENG_commons_seats^i
			add_to_variable = { ENG_commons_total_seats = ENG_commons_seats^i }
		}
	}

	# Compute combined popularity of the top two parties for proportional regional splits
	set_variable = { ENG_commons_p1p2_total = ENG_commons_pop1 }
	add_to_variable = { ENG_commons_p1p2_total = ENG_commons_pop2 }

	# --- Scotland: 59 seats ---
	# ENG_scottish_nationalists (0-100) determines the SNP bloc's share.
	# The remaining seats are split between the top two parties in proportion to their popularity.
	set_temp_variable = { scot_nat_temp = ENG_scottish_nationalists }
	multiply_temp_variable = { scot_nat_temp = 0.59 }
	set_variable = { ENG_commons_seats_SNP = scot_nat_temp }
	round_variable = ENG_commons_seats_SNP
	# Random electoral swing: ±3 seats for Scotland
	random_list = {
		10 = { add_to_variable = { ENG_commons_seats_SNP = 3 } }
		20 = { add_to_variable = { ENG_commons_seats_SNP = 2 } }
		20 = { add_to_variable = { ENG_commons_seats_SNP = 1 } }
		0 = { }
		20 = { subtract_from_variable = { ENG_commons_seats_SNP = 1 } }
		20 = { subtract_from_variable = { ENG_commons_seats_SNP = 2 } }
		10 = { subtract_from_variable = { ENG_commons_seats_SNP = 3 } }
	}
	clamp_variable = { var = ENG_commons_seats_SNP min = 0 max = 59 }
	add_to_variable = { ENG_commons_total_seats = ENG_commons_seats_SNP }

	set_variable = { ENG_commons_scot_remaining = 59 }
	subtract_from_variable = { ENG_commons_scot_remaining = ENG_commons_seats_SNP }

	if = {
		limit = { check_variable = { ENG_commons_p1p2_total > 0 } }
		set_temp_variable = { scot_p1_temp = ENG_commons_pop1 }
		divide_temp_variable = { scot_p1_temp = ENG_commons_p1p2_total }
		multiply_temp_variable = { scot_p1_temp = ENG_commons_scot_remaining }
		set_variable = { ENG_commons_scot_p1_share = scot_p1_temp }
		round_variable = ENG_commons_scot_p1_share
		add_to_variable = { ENG_commons_seats^ENG_commons_idx1 = ENG_commons_scot_p1_share }
		add_to_variable = { ENG_commons_total_seats = ENG_commons_scot_p1_share }

		set_variable = { ENG_commons_scot_p2_share = ENG_commons_scot_remaining }
		subtract_from_variable = { ENG_commons_scot_p2_share = ENG_commons_scot_p1_share }
		add_to_variable = { ENG_commons_seats^ENG_commons_idx2 = ENG_commons_scot_p2_share }
		add_to_variable = { ENG_commons_total_seats = ENG_commons_scot_p2_share }
	}

	# --- Wales: 40 seats ---
	# ENG_wales_nationalists (0-100) determines the Plaid Cymru bloc's share.
	# The remaining seats are split between the top two parties in proportion to their popularity.
	set_temp_variable = { wales_nat_temp = ENG_wales_nationalists }
	multiply_temp_variable = { wales_nat_temp = 0.40 }
	set_variable = { ENG_commons_seats_Plaid = wales_nat_temp }
	round_variable = ENG_commons_seats_Plaid
	# Random electoral swing: ±2 seats for Wales
	random_list = {
		10 = { add_to_variable = { ENG_commons_seats_Plaid = 2 } }
		25 = { add_to_variable = { ENG_commons_seats_Plaid = 1 } }
		30 = { }
		25 = { subtract_from_variable = { ENG_commons_seats_Plaid = 1 } }
		10 = { subtract_from_variable = { ENG_commons_seats_Plaid = 2 } }
	}
	clamp_variable = { var = ENG_commons_seats_Plaid min = 0 max = 40 }
	add_to_variable = { ENG_commons_total_seats = ENG_commons_seats_Plaid }

	set_variable = { ENG_commons_wales_remaining = 40 }
	subtract_from_variable = { ENG_commons_wales_remaining = ENG_commons_seats_Plaid }

	if = {
		limit = { check_variable = { ENG_commons_p1p2_total > 0 } }
		set_temp_variable = { wales_p1_temp = ENG_commons_pop1 }
		divide_temp_variable = { wales_p1_temp = ENG_commons_p1p2_total }
		multiply_temp_variable = { wales_p1_temp = ENG_commons_wales_remaining }
		set_variable = { ENG_commons_wales_p1_share = wales_p1_temp }
		round_variable = ENG_commons_wales_p1_share
		add_to_variable = { ENG_commons_seats^ENG_commons_idx1 = ENG_commons_wales_p1_share }
		add_to_variable = { ENG_commons_total_seats = ENG_commons_wales_p1_share }

		set_variable = { ENG_commons_wales_p2_share = ENG_commons_wales_remaining }
		subtract_from_variable = { ENG_commons_wales_p2_share = ENG_commons_wales_p1_share }
		add_to_variable = { ENG_commons_seats^ENG_commons_idx2 = ENG_commons_wales_p2_share }
		add_to_variable = { ENG_commons_total_seats = ENG_commons_wales_p2_share }
	}

	# --- Northern Ireland: 18 seats ---
	# ENG_north_irish_nationalists (0-100): 100 = fully Sinn Fein/republican.
	# The unionist bloc (DUP, UUP, TUV, etc.) holds the remainder.
	set_temp_variable = { nir_rep_temp = ENG_north_irish_nationalists }
	multiply_temp_variable = { nir_rep_temp = 0.18 }
	set_variable = { ENG_commons_seats_NIR_republican = nir_rep_temp }
	round_variable = ENG_commons_seats_NIR_republican
	# Random electoral swing: ±2 seats for Northern Ireland
	random_list = {
		10 = { add_to_variable = { ENG_commons_seats_NIR_republican = 2 } }
		25 = { add_to_variable = { ENG_commons_seats_NIR_republican = 1 } }
		30 = { }
		25 = { subtract_from_variable = { ENG_commons_seats_NIR_republican = 1 } }
		10 = { subtract_from_variable = { ENG_commons_seats_NIR_republican = 2 } }
	}
	clamp_variable = { var = ENG_commons_seats_NIR_republican min = 0 max = 18 }

	set_variable = { ENG_commons_seats_NIR_unionist = 18 }
	subtract_from_variable = { ENG_commons_seats_NIR_unionist = ENG_commons_seats_NIR_republican }

	add_to_variable = { ENG_commons_total_seats = ENG_commons_seats_NIR_republican }
	add_to_variable = { ENG_commons_total_seats = ENG_commons_seats_NIR_unionist }

	# --- Independents: buffer to always sum to 650 ---
	# Rounding across regions will leave a small remainder; independents absorb it.
	set_variable = { ENG_commons_seats_INDEPENDENT = 650 }
	subtract_from_variable = { ENG_commons_seats_INDEPENDENT = ENG_commons_total_seats }
	if = {
		limit = { check_variable = { ENG_commons_seats_INDEPENDENT < 0 } }
		set_variable = { ENG_commons_seats_INDEPENDENT = 0 }
	}
	add_to_variable = { ENG_commons_total_seats = ENG_commons_seats_INDEPENDENT }

	ENG_commons_restore_parties = yes
}
