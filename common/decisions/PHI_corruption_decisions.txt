# Philippines Corruption System - Decisions
# All decisions reduce PHI_corruption_influence and are visible only for the Philippines.
# Variables used:
#   PHI_corruption_influence  (0–100, current corruption level)
#   PHI_corruption_change     (temp variable, negative = reduce corruption)

PHI_corruption_decision_category = {

	# ─── 1. Anti-Corruption Propaganda ───
	# Cheap, always available. Small reduction over time.
	PHI_anti_corruption_propaganda = {
		icon = GFX_decision_category_generic_propaganda

		available = {
			is_corruption_nation = yes
		}

		modifier = {
			political_power_factor = -0.05
			stability_weekly = 0.002
		}

		days_remove = 60
		days_re_enable = 30

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision PHI_anti_corruption_propaganda"
			hidden_effect = {
				set_country_flag = { flag = PHI_anti_corruption_propaganda_active value = 1 days = 60 }
			}
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove PHI_anti_corruption_propaganda"
			set_temp_variable = { PHI_corruption_change = -3 }
			PHI_modify_corruption_variables_effect = yes
			add_stability = 0.02
		}

		ai_will_do = {
			base = 10
			modifier = {
				factor = 5
				check_variable = { PHI_corruption_influence > 49 }
			}
		}
	}

	# ─── 2. Strengthen Police Integrity ───
	# Reduces corruption through internal affairs.
	PHI_strengthen_police_integrity = {
		icon = GFX_decision_category_generic_political_actions

		available = {
			is_corruption_nation = yes
			NOT = { has_decision = PHI_deploy_military_integrity_units }
			command_power > 24
		}

		cost = 50
		modifier = {
			political_power_factor = -0.05
			police_cost_multiplier_modifier = 0.10
			max_command_power = -25
		}

		days_remove = 60
		days_re_enable = 30

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision PHI_strengthen_police_integrity"
			add_command_power = -25
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove PHI_strengthen_police_integrity"
			set_temp_variable = { PHI_corruption_change = -4 }
			PHI_modify_corruption_variables_effect = yes
		}

		ai_will_do = {
			base = 5
			modifier = {
				factor = 2.5
				check_variable = { PHI_corruption_influence > 49 }
			}
			modifier = {
				factor = 0
				has_political_power < 100
			}
		}
	}

	# ─── 3. Deploy Military Integrity Units ───
	# More aggressive than police, higher cost, larger reduction.
	PHI_deploy_military_integrity_units = {
		icon = GFX_decision_army_dec

		available = {
			is_corruption_nation = yes
			NOT = { has_decision = PHI_strengthen_police_integrity }
			command_power > 49
			NOT = { has_idea = no_military }
		}

		cost = 50
		modifier = {
			political_power_factor = -0.05
			personnel_cost_multiplier_modifier = 0.10
			max_command_power = -50
		}

		days_remove = 60
		days_re_enable = 30

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision PHI_deploy_military_integrity_units"
			add_command_power = -50
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove PHI_deploy_military_integrity_units"
			set_temp_variable = { PHI_corruption_change = -6 }
			PHI_modify_corruption_variables_effect = yes
		}

		ai_will_do = {
			base = 0
		}
	}

	# ─── 4. Asset Declaration Law ───
	# Long-term structural reform. Large reduction, stability bonus.
	PHI_asset_declaration_law = {
		icon = GFX_decision_category_generic_political_actions

		available = {
			is_corruption_nation = yes
			NOT = { has_decision = PHI_asset_declaration_law }
		}

		cost = 50

		days_remove = 180
		days_re_enable = 90

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision PHI_asset_declaration_law"
			custom_effect_tooltip = PHI_asset_declaration_law_active_tt
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove PHI_asset_declaration_law"
			set_temp_variable = { PHI_corruption_change = -6 }
			PHI_modify_corruption_variables_effect = yes
			add_stability = 0.05
		}

		ai_will_do = {
			base = 8
			modifier = {
				factor = 0
				has_political_power < 100
			}
		}
	}

	# ─── 5. Prosecute Corrupt Officials ───
	# Random outcome: success or failure. Risk/reward decision.
	PHI_prosecute_corrupt_officials = {
		icon = GFX_decision_category_cartel

		available = {
			is_corruption_nation = yes
		}

		cost = 50
		fixed_random_seed = no

		days_remove = 90
		days_re_enable = 60

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision PHI_prosecute_corrupt_officials"
			custom_effect_tooltip = PHI_prosecute_corrupt_officials_tt
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove PHI_prosecute_corrupt_officials"
			random_list = {
				35 = {
					# Success: officials convicted
					modifier = {
						factor = 1.30
						OR = {
							has_idea = police_03
							has_idea = police_04
							has_idea = police_05
						}
					}
					modifier = {
						factor = 1.40
						difficulty < 2
					}
					custom_effect_tooltip = PHI_prosecution_success_tt
					set_temp_variable = { PHI_corruption_change = -8 }
					PHI_modify_corruption_variables_effect = yes
					add_stability = 0.03
					set_party_index_to_ruling_party = yes
					set_temp_variable = { party_popularity_increase = 0.02 }
					add_relative_party_popularity = yes
				}
				65 = {
					# Failure: officials walk free, backlash
					modifier = {
						factor = 1.50
						check_variable = { PHI_corruption_influence > 49 }
					}
					modifier = {
						factor = 1.25
						difficulty > 2
					}
					custom_effect_tooltip = PHI_prosecution_failure_tt
					set_temp_variable = { PHI_corruption_change = 3 }
					PHI_modify_corruption_variables_effect = yes
					add_stability = -0.02
					set_party_index_to_ruling_party = yes
					set_temp_variable = { party_popularity_increase = -0.03 }
					add_relative_party_popularity = yes
				}
			}
		}

		ai_will_do = {
			base = 5
			modifier = {
				factor = 0
				has_political_power < 100
			}
		}
	}

	# ─── 6. Audit Government Contracts ───
	# Targets procurement fraud. Treasury gain on success, PP loss on failure.
	PHI_audit_government_contracts = {
		icon = GFX_decision_category_generic_industry

		available = {
			is_corruption_nation = yes
		}

		cost = 50
		fixed_random_seed = no

		days_remove = 90
		days_re_enable = 30

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision PHI_audit_government_contracts"
			custom_effect_tooltip = PHI_audit_government_contracts_tt
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove PHI_audit_government_contracts"
			random_list = {
				40 = {
					# Success: Recovered funds + corruption reduced
					modifier = {
						factor = 1.25
						check_variable = { PHI_corruption_influence < 30 }
					}
					modifier = {
						factor = 1.40
						difficulty < 2
					}
					custom_effect_tooltip = PHI_audit_success_tt
					set_temp_variable = { PHI_corruption_change = -5 }
					PHI_modify_corruption_variables_effect = yes
					set_temp_variable = { treasury_change = gdp_total }
					multiply_temp_variable = { treasury_change = 0.008 }
					modify_treasury_effect = yes
				}
				60 = {
					# Failure: Tipped-off officials, political fallout
					modifier = {
						factor = 1.50
						check_variable = { PHI_corruption_influence > 49 }
					}
					modifier = {
						factor = 1.25
						difficulty > 2
					}
					custom_effect_tooltip = PHI_audit_failure_tt
					set_temp_variable = { PHI_corruption_change = 2 }
					PHI_modify_corruption_variables_effect = yes
					add_political_power = -25
				}
			}
		}

		ai_will_do = {
			base = 7
			modifier = {
				factor = 0
				has_political_power < 100
			}
		}
	}

	# ─── 7. International Anti-Corruption Cooperation ───
	# Long treaty, big reduction, Western outlook bonus. Expensive.
	PHI_international_anticorruption_cooperation = {
		icon = GFX_decision_category_generic_foreign_policy

		available = {
			is_corruption_nation = yes
			check_variable = { domestic_influence_amount > 50 }
		}

		cost = 75

		days_remove = 365
		days_re_enable = 180

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision PHI_international_anticorruption_cooperation"
			custom_effect_tooltip = PHI_intl_cooperation_active_tt
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove PHI_international_anticorruption_cooperation"
			set_temp_variable = { PHI_corruption_change = -10 }
			PHI_modify_corruption_variables_effect = yes
			add_stability = 0.05
			add_political_power = 25
		}

		ai_will_do = {
			base = 3
			modifier = {
				factor = 3
				check_variable = { PHI_corruption_influence > 59 }
			}
			modifier = {
				factor = 0
				has_political_power < 150
			}
		}
	}

}
