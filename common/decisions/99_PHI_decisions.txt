PHI_edsa_dos_category = {

	# Timed mission — activates automatically in October 2000.
	# Applies the initial scandal effects on activation via complete/timeout.
	# Cancelled if the player takes PHI_open_the_second_envelope first.
	PHI_the_people_grow_disillusioned = {
		icon = political_actions
		is_good = no

		visible = {
			original_tag = PHI
			date > 2000.10.1
			NOT = { has_country_flag = PHI_edsa_dos_resolved }
		}

		available = {
			original_tag = PHI
			date > 2000.10.1
			NOT = { has_country_flag = PHI_edsa_dos_resolved }
		}

		activation = {
			original_tag = PHI
			date > 2000.10.1
			NOT = { has_country_flag = PHI_edsa_dos_resolved }
		}

		cancel_if_not_visible = yes
		selectable_mission = yes
		days_mission_timeout = 110

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete PHI_the_people_grow_disillusioned"
			set_temp_variable = { temp_change = 8 }
			PHI_change_the_strength_of_corruption = yes
			add_stability = -0.04
			add_timed_idea = { idea = PHI_edsa_dos_impeachment_atmosphere days = 120 }
			add_stability = -0.05
			add_political_power = -50
			country_event = { id = PHI.2 days = 1 }
		}

		timeout_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision timeout PHI_the_people_grow_disillusioned"
			set_temp_variable = { temp_change = 8 }
			PHI_change_the_strength_of_corruption = yes
			add_stability = -0.04
			add_timed_idea = { idea = PHI_edsa_dos_impeachment_atmosphere days = 120 }
			add_stability = -0.05
			add_political_power = -50
			country_event = { id = PHI.2 days = 1 }
		}

		ai_will_do = { base = 0 }
	}
