		context_type = player_context

		dirty = tutorial_gui_dirty_var

		visible = {
			is_ai = no
		}

		effects = {
			open_tutorial_window_button_click = {
				if = {
					limit = { NOT = { has_country_flag = open_tutorial_window } }

					if = {
						limit = {
							NOT = {
								OR = {
									has_country_flag = tutorial_tab_overview
									has_country_flag = tutorial_tab_economy
									has_country_flag = tutorial_tab_politics
									has_country_flag = tutorial_tab_influence
									has_country_flag = tutorial_tab_research
									has_country_flag = tutorial_tab_military
									has_country_flag = tutorial_tab_advanced
								}
							}
						}
						set_country_flag = tutorial_tab_overview
					}
				}
				else = {
					clr_country_flag = open_tutorial_window
				}
				add_to_variable = { tutorial_gui_dirty_var = 1 }
			}
		}
	}
