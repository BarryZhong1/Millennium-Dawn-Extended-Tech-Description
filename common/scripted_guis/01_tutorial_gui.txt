			open_tutorial_window_button_click = {
				if = {
					limit = { NOT = { has_country_flag = open_tutorial_window } }

					if = {
						limit = {
							NOT = {
								OR = {
									has_country_flag = tutorial_cat_overview

									has_country_flag = tutorial_cat_politics
									has_country_flag = tutorial_cat_military
									has_country_flag = tutorial_cat_diplomacy
								}
							}
						}
						set_md_tutorial_overview_default = yes
					}

					if = {
						limit = {
							NOT = {
								OR = {
									has_country_flag = tutorial_topic_overview_intro
									has_country_flag = tutorial_topic_overview_systems
									has_country_flag = tutorial_topic_overview_power
									has_country_flag = tutorial_topic_overview_playloop
									has_country_flag = tutorial_topic_overview_ui
									has_country_flag = tutorial_topic_economy_budget
									has_country_flag = tutorial_topic_economy_gdp
									has_country_flag = tutorial_topic_economy_taxation
									has_country_flag = tutorial_topic_economy_debt
									has_country_flag = tutorial_topic_economy_energy
									has_country_flag = tutorial_topic_politics_ideologies
									has_country_flag = tutorial_topic_politics_parties
									has_country_flag = tutorial_topic_politics_elections
									has_country_flag = tutorial_topic_politics_coalitions
									has_country_flag = tutorial_topic_politics_laws
									has_country_flag = tutorial_topic_military_modernization

									has_country_flag = tutorial_topic_military_resources
									has_country_flag = tutorial_topic_military_missiles
									has_country_flag = tutorial_topic_military_contractors
									has_country_flag = tutorial_topic_diplomacy_influence
									has_country_flag = tutorial_topic_diplomacy_relations
									has_country_flag = tutorial_topic_diplomacy_trade
									has_country_flag = tutorial_topic_diplomacy_organizations
									has_country_flag = tutorial_topic_diplomacy_satellites
								}
							}
						}
						if = {
							limit = { has_country_flag = tutorial_cat_economy }
							select_md_tutorial_economy_budget = yes
						}
						else_if = {
							limit = { has_country_flag = tutorial_cat_politics }
							select_md_tutorial_politics_ideologies = yes
						}
						else_if = {
							limit = { has_country_flag = tutorial_cat_military }
							select_md_tutorial_military_modernization = yes
						}
						else_if = {
							limit = { has_country_flag = tutorial_cat_diplomacy }
							select_md_tutorial_diplomacy_influence = yes
						}
						else = {
							select_md_tutorial_overview_intro = yes
						}
					}
				}
				else = {
					clr_country_flag = open_tutorial_window
				}
				add_to_variable = { tutorial_gui_dirty_var = 1 }
			}
		}
	}
