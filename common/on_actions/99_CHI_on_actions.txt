
on_actions = {
	on_unit_leader_level_up = {
		effect = {
			if = {
				limit = {
					FROM = { original_tag = CHI }
					OR = {
						FROM = {
							has_country_flag = {
								flag = CHI_cnc_general_quote_used
								days > 30
							}
							NOT = { has_country_flag = CHI_cnc_general_quote_used }
						}
					}
				}
				ROOT = {
					if = {
						limit = { is_character = CHI_general_fai }
						FROM = {
							sound_effect = "CHI_general_fai_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
					else_if = {
						limit = { is_character = CHI_general_kwai  }
						FROM = {
							sound_effect = "CHI_general_kwai_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
					else_if = {
						limit = { is_character = CHI_general_leang }
						FROM = {
							sound_effect = "CHI_general_leang_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
					else_if = {
						limit = { is_character = CHI_general_tao }
						FROM = {
							sound_effect = "CHI_general_tao_shout"
							clr_country_flag = CHI_cnc_general_quote_used
							set_country_flag = CHI_cnc_general_quote_used
						}
					}
				}
			}
		}
	}
}
