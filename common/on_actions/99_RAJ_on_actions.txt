on_actions = {
	on_monthly_RAJ = {
		random_events = {
			1000 = 0

			170 = RAJ_naxal.11
			80 = RAJ_naxal.13
			60 = RAJ_naxal.14
		}

		effect = {
			india_religious_events = yes
			RAJ_check_religious_harmony = yes
			add_to_variable = { RAJ_brics_gui_dirty = 1 }

			random = {
				chance = 20
				subtract_from_variable = { RAJ.kasmir_readiness = 5 }
				clamp_variable = { var = RAJ.kasmir_readiness min = 0 max = 100 }
			}
			random = {
				chance = 20
				subtract_from_variable = { RAJ.kasmir_tension = 5 }
				clamp_variable = { var = RAJ.kasmir_tension min = 0 max = 100 }
			}

			random = {
				chance = 20
				subtract_from_variable = { RAJ.kasmir_china_readiness = 5 }
				clamp_variable = { var = RAJ.kasmir_china_readiness min = 0 max = 100 }
			}
			random = {
				chance = 20
				subtract_from_variable = { GLOBAL.kasmir_tension_china = 5 }
				clamp_variable = { var = GLOBAL.kasmir_tension_china min = 0 max = 100 }
			}

			random = {
				chance = 20
				subtract_from_variable = { RAJ.arunachal_readiness = 5 }
				clamp_variable = { var = RAJ.arunachal_readiness min = 0 max = 100 }
			}

			# Pre-compute naxalite state flags for decision visible blocks (avoids per-frame any_controlled_state)
			if = {
				limit = {
					any_controlled_state = { RAJ_state_has_any_naxalite = yes }
				}
				set_country_flag = RAJ_has_naxalite_states
			}
			else = {
				clr_country_flag = RAJ_has_naxalite_states
			}
			if = {
				limit = { has_country_flag = RAJ_has_naxalite_states }
				if = {
					limit = {
						any_controlled_state = {
							OR = {
								has_dynamic_modifier = { modifier = RAJ_naxalite_insurgency }
								has_dynamic_modifier = { modifier = RAJ_naxalite_insurgency_1 }
							}
						}
					}
					set_country_flag = RAJ_has_severe_naxalite_states
				}
				else = {
					clr_country_flag = RAJ_has_severe_naxalite_states
				}
			}
			else = {
				clr_country_flag = RAJ_has_severe_naxalite_states
			}

			# Naxal-Free Milestone (fires when last naxalite state is cleared)
			if = {
				limit = {
					NOT = { has_country_flag = RAJ_naxal_free_achieved }
					NOT = { has_country_flag = RAJ_has_naxalite_states }
					check_variable = { RAJ_strength_of_maoists < 5 }
				}
				RAJ = { country_event = RAJ_naxal.16 }
			}
		}
	}
	# Religious breakaway mechanic moved to mission decisions in 99_RAJ_religious_missions.txt
}
