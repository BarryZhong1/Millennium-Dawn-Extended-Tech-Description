	on_startup = {
		effect = {
			# add companies to countries
			POL = {
				set_temp_variable = { cac_selected_company = 9 }
				set_temp_variable = { cac_company_type = 0 }
				CAC_add_company = yes

				set_temp_variable = { cac_selected_company = 10 }
				set_temp_variable = { cac_company_type = 1 }
				CAC_add_company = yes

				set_temp_variable = { cac_selected_company = 11 }
				set_temp_variable = { cac_company_type = 2 }
				CAC_add_company = yes

				set_temp_variable = { cac_selected_company = 12 }
				set_temp_variable = { cac_company_type = 3 }
				CAC_add_company = yes

				set_temp_variable = { cac_selected_company = 13 }
				set_temp_variable = { cac_company_type = 4 }
				CAC_add_company = yes

				set_temp_variable = { cac_selected_company = 14 }
				set_temp_variable = { cac_company_type = 5 }
				CAC_add_company = yes

				set_temp_variable = { cac_selected_company = 15 }
				set_temp_variable = { cac_company_type = 6 }
				CAC_add_company = yes

				set_temp_variable = { cac_selected_company = 16 }
				set_temp_variable = { cac_company_type = 7 }
				CAC_add_company = yes

				set_temp_variable = { cac_selected_company = 17 }
				set_temp_variable = { cac_company_type = 8 }
				CAC_add_company = yes
			}


			# GLOBAL VARIABLES

			# ceo
			for_loop_effect = {
				start = 0
				end = 4
				value = i

				add_to_array = {
					array = CAC_available_ceo_list
					value = i
				}
			}

			every_country = {
				# SETUP VARIABLES #

				# foreign companies
				set_variable = { CAC_foreign_companies_limit = 6 }

				# ui variables
				set_variable = { opened_company_details = -1 }
				set_variable = { companies_history_tab = 2 }
				set_variable = { companies_upgrades_tab = 1 }

				# generic industries
				if = {
					limit = {
						check_variable = { CAC_companies^num < 1 }
					}

					set_temp_variable = { cac_selected_company = 0 }
					set_temp_variable = { cac_company_type = 0 }
					CAC_add_company = yes

					set_temp_variable = { cac_selected_company = 1 }
					set_temp_variable = { cac_company_type = 1 }
					CAC_add_company = yes

					set_temp_variable = { cac_selected_company = 2 }
					set_temp_variable = { cac_company_type = 2 }
					CAC_add_company = yes

					set_temp_variable = { cac_selected_company = 3 }
					set_temp_variable = { cac_company_type = 3 }
					CAC_add_company = yes

					set_temp_variable = { cac_selected_company = 4 }
					set_temp_variable = { cac_company_type = 4 }
					CAC_add_company = yes

					set_temp_variable = { cac_selected_company = 5 }
					set_temp_variable = { cac_company_type = 5 }
					CAC_add_company = yes

					set_temp_variable = { cac_selected_company = 6 }
					set_temp_variable = { cac_company_type = 6 }
					CAC_add_company = yes

					set_temp_variable = { cac_company_type = 7 }
					CAC_add_company = yes

					set_temp_variable = { cac_selected_company = 8 }
					set_temp_variable = { cac_company_type = 8 }
					CAC_add_company = yes
				}

				add_dynamic_modifier = { modifier = CAC_companies_dynamic_modifiers }
				add_dynamic_modifier = { modifier = CAC_foreign_companies_dynamic_modifiers }

				CAC_update_variables = yes
			}
		}
	}
	on_weekly = {
		effect = {
			CAC_update_upkeep = yes
		}
	}
	on_monthly = {
		effect = {
			CAC_change_influence = yes
		}
	}

	on_declare_war = {
		effect = {
			for_each_loop = {
				array = ROOT.CAC_foreign_companies
				index = index
				break = break

				if = {
					limit = { check_variable = { ROOT.CAC_foreign_companies_countries^index = THIS } }

					remove_from_array = {
						array = ROOT.CAC_foreign_companies
						index = index
					}
					remove_from_array = {
						array = ROOT.CAC_foreign_companies_countries
						index = index
					}

					set_temp_variable = { cac_type = THIS.CAC_companies_type^i }
					ROOT = { CAC_remove_bonuses_from_foreign_company_effect = yes }

					THIS = {
						set_variable = { CAC_companies_income_modifier^i = -0.1 }

						set_temp_variable = { cac_slot = v }
						set_temp_variable = { cac_additional_income_change = -0.1 }
						CAC_modify_additional_income_effect = yes
					}
				}
			}

			for_each_loop = {
				array = FROM.CAC_foreign_companies
				index = index
				break = break

				if = {
					limit = { check_variable = { FROM.CAC_foreign_companies_countries^index = ROOT } }

					remove_from_array = {
						array = FROM.CAC_foreign_companies
						index = index
					}
					remove_from_array = {
						array = FROM.CAC_foreign_companies_countries
						index = index
					}

					set_temp_variable = { cac_type = ROOT.CAC_companies_type^i }
					FROM = { CAC_remove_bonuses_from_foreign_company_effect = yes }

					FROM = {
						set_variable = { CAC_companies_income_modifier^i = -0.1 }

						set_temp_variable = { cac_slot = v }
						set_temp_variable = { cac_additional_income_change = -0.1 }
						CAC_modify_additional_income_effect = yes
					}
				}
			}
		}
	}
}
