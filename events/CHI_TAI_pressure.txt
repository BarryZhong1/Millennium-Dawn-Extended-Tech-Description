# Event 1: Economic Pressure — China opens the campaign
country_event = {
	id = chi_tai_pressure.1
	title = chi_tai_pressure.1.t
	desc = chi_tai_pressure.1.d
	picture = GFX_report_event_china_diplomacy

	is_triggered_only = yes

	option = {
		name = chi_tai_pressure.1.a
		log = "[GetDateText]: [This.GetName]: chi_tai_pressure.1.a executed"
		# Aggressive trade leverage — higher short-term impact, risk of retaliation
		custom_effect_tooltip = CHI_tai_pressure_economic_tt
		hidden_effect = {
			set_country_flag = CHI_TAI_pressure_aggressive
			USA = { country_event = { id = chi_tai_pressure.2  days = 14 } }
		}
		add_opinion_modifier = { target = USA  modifier = chi_usa_pressure_campaign }
		ai_chance = {
			base = 6
			modifier = {
				add = 3
				has_global_flag = CHI_ONE_SYSTEM_PATH
			}
		}
	}

	option = {
		name = chi_tai_pressure.1.b
		log = "[GetDateText]: [This.GetName]: chi_tai_pressure.1.b executed"
		# Measured diplomatic pressure — lower risk, slower progress
		custom_effect_tooltip = CHI_tai_pressure_diplomatic_tt
		hidden_effect = {
			USA = { country_event = { id = chi_tai_pressure.2  days = 21 } }
		}
		add_opinion_modifier = { target = USA  modifier = chi_usa_pressure_campaign }
		ai_chance = {
			base = 4
		}
	}
}

# Event 2: US Internal Debate — US decides how to respond to Chinese pressure

		CHI = { add_opinion_modifier = { target = USA  modifier = tai_diplomatic_isolation } }
		TAI = { add_opinion_modifier = { target = USA  modifier = reduced_tensions } }
		ai_chance = {
			base = 8
			modifier = {
				add = 5
				has_opinion = { target = CHI  value < -10 }
			}
			modifier = {
				add = 4
				TAI = { is_in_faction_with = USA }
			}
			modifier = {
				factor = 0

		ai_chance = {
			base = 4

			}

# Event 3: China escalates — Diplomatic isolation + military signalling
country_event = {
	id = chi_tai_pressure.3
	title = chi_tai_pressure.3.t
	desc = chi_tai_pressure.3.d
	picture = GFX_report_event_china_military

	is_triggered_only = yes

	# Launch full diplomatic isolation campaign
	option = {
		name = chi_tai_pressure.3.a
		log = "[GetDateText]: [This.GetName]: chi_tai_pressure.3.a executed"
		add_war_support = 0.03
		# Build coalition: countries with low US opinion apply pressure
		every_other_country = {
			limit = {
				has_opinion = { target = USA  value < -15 }
				NOT = { is_in_faction_with = USA }
				is_major = yes
			}
			add_opinion_modifier = { target = CHI  modifier = shared_future_community }
		}
		hidden_effect = {
			USA = { country_event = { id = chi_tai_pressure.5  days = 21 } }
		}
		ai_chance = { base = 8 }
	}

	# Conduct Taiwan Strait military exercises — direct pressure
	option = {
		name = chi_tai_pressure.3.b
		log = "[GetDateText]: [This.GetName]: chi_tai_pressure.3.b executed"
		add_war_support = 0.05
		add_threat = 0.5
		hidden_effect = {
			USA = { country_event = { id = chi_tai_pressure.5  days = 14 } }
		}
		ai_chance = {
			base = 4
			modifier = {
				add = 4
				has_global_flag = CHI_ONE_SYSTEM_PATH
			}
		}
	}
}
