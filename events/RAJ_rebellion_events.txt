add_namespace = RAJ_rebellion

# Ladakh Rebellion (Muslim opinion)
country_event = {
	id = RAJ_rebellion.1
	title = RAJ_rebellion.1.t
	desc = RAJ_rebellion.1.d
	picture = GFX_musl_rally2
	is_triggered_only = yes

	option = {
		name = RAJ_rebellion.1.a
		trigger = {
			owns_state = 993
			check_variable = { RAJ_muslim_opinion < 5 }
		}
		log = "[GetDateText]: [Root.GetName]: RAJ_rebellion.1.a executed"
		set_country_flag = RAJ_ladakh_rebellion_fired
		hidden_effect = {
			create_dynamic_country = {
				original_tag = RAJ
				set_country_flag = RAJ_ladakh_rebellion
				copy_tag = RAJ
				ingame_startup_nations_effect = yes
				set_cosmetic_tag = LAD
				set_country_flag = is_civil_war_faction
				set_popularities = {
					nationalist = 80
					neutrality = 20
				}
				set_politics = {
					ruling_party = nationalist
					elections_allowed = no
				}
				993 = {
					set_demilitarized_zone = no
					transfer_state_to = PREV
				}
			}
		}
		add_stability = -0.10
		add_war_support = -0.05
	}

	option = {
		name = RAJ_rebellion.1.b
		trigger = {
			OR = {
				NOT = { owns_state = 993 }
				check_variable = { RAJ_muslim_opinion > 4 }
			}
		}
		log = "[GetDateText]: [Root.GetName]: RAJ_rebellion.1.b executed (crisis averted)"
		if = {
			limit = { owns_state = 993 }
			993 = { set_demilitarized_zone = no }
		}
		add_stability = 0.02
	}
}
