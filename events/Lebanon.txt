# Route to parliament — choose the path
country_event = {
	id = Leb.storyevent.21
	title = Leb.storyevent.21.t
	desc = Leb.storyevent.21.d

	is_triggered_only = yes

	option = {
		name = Leb.storyevent.21.a
		log = "[GetDateText]: [This.GetName]: Leb.storyevent.21.a executed"
		if = {
			limit = {
				check_variable = {
					LEB_hariri_network_strength > 0.49
				}
			}
			hidden_effect = {
				country_event = {
					id = Leb.storyevent.25
					days = 2
				}
			}
		}
		else = {
			hidden_effect = {
				country_event = {
					id = Leb.storyevent.22
					days = 2
				}
			}
		}
	}
	option = {
		name = Leb.storyevent.21.b
		log = "[GetDateText]: [This.GetName]: Leb.storyevent.21.b executed"
		hidden_effect = {
			country_event = {
				id = Leb.storyevent.24
				days = 10
			}
		}
		add_to_variable = {
			var = LEB_public_support
			value = 0.1
			tooltip = LEB_public_support_add_10_TT
		}
	}
}
